






















const mongoose = require('mongoose');

const productSchema = new mongoose.Schema({
    brand:{
        type: String,
        required: true
    },
    name:{
        type: String,
        required: true,
        validate: {
            validator: async function (name) {
                const count = await mongoose.models.Product.countDocuments({ name: name, brand: this.brand });
                return count === 0;
            },
            message: props => `${props.value} already exists for the given brand`
        }

    },
    price: {
        type: Number,
        required: true
    },    
    itemqty: {
        type: Number,
        required: true
    },
    addedAt:{
        type: Date,
        default: ()=> Date.now(),
    },
    updatedAt: Date,
    description: String
});

productSchema.pre('save', function(next){
    this.updatedAt = Date.now();
    next();
});





const product = mongoose.model('product', productSchema);





module.exports = {
    product: product,
    // key2: value2,
  };
